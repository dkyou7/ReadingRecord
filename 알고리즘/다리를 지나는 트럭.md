# 다리를 지나는 트럭
def solution(bridge_length, weight, truck_weights):
    time = 0
    q = bridge_length*[0]
    while q:
        time += 1
        q.pop(0)
        if truck_weights:
            if sum(q) + truck_weights[0] <= weight:
                q.append(truck_weights.pop(0))
            else:
                q.append(0)
    return time
